    /**
     * 更改颜色透明度的方法
     *
     * @param color 颜色
     * @param alpha 新的透明度
     * @return 返回一个具有新透明度的颜色
     */
    private int changeColorAlpha(int color, int alpha) {
        int r = (color >>> 16) & 0xFF;
        int g = (color >>> 8) & 0xFF;
        int b = (color) & 0xFF;
        return alpha << 24 | r << 16 | g << 8 | b;
    }
